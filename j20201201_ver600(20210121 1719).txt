--  DDL for View FINISH_BID
--------------------------------------------------------

  CREATE OR REPLACE FORCE VIEW "J20201201"."FINISH_BID" ("B_NUM", "B_EXPIRATION", "BID_PRICE", "USER_EMAIL", "SELLER_EMAIL") AS 
  SELECT b.b_num, b.b_expiration, bid.bid_price, bid.user_email, b.seller_email FROM BID bid, BOARD b WHERE bid.b_num(+) = b.b_num ;
--------------------------------------------------------
--  DDL for View FINISH_BID1
--------------------------------------------------------

  CREATE OR REPLACE FORCE VIEW "J20201201"."FINISH_BID1" ("B_NUM", "SELLER_EMAIL", "BID_PRICE", "USER_EMAIL") AS 
  SELECT fb.b_num, fb.seller_email, fb.bid_price, fb.user_email FROM finish_bid fb WHERE fb.b_expiration < TO_CHAR(SYSDATE,'YYYY/MM/DD hh24:mi:ss') AND NOT EXISTS (SELECT 1 FROM BID_DONE bd WHERE fb.b_num=bd.b_num) ;
--------------------------------------------------------
--  DDL for View NOW_BID
--------------------------------------------------------

  CREATE OR REPLACE FORCE VIEW "J20201201"."NOW_BID" ("B_NUM", "B_EXPIRATION", "PD_PRICE", "PD_UNIT") AS 
  SELECT b.b_num, b.b_expiration, pi.pd_price, pi.pd_unit FROM PRODUCT_INFO pi, BOARD b WHERE b.pd_num=pi.pd_num ;
--------------------------------------------------------
--  DDL for View PAY_BACK
--------------------------------------------------------

  CREATE OR REPLACE FORCE VIEW "J20201201"."PAY_BACK" ("BID_NUM", "USER_EMAIL", "BALANCE") AS 
  SELECT am.bid_num, am.user_email, am.balance FROM BID b, AUCTION_MONEY am WHERE am.bid_num = b.bid_num(+) AND b_num = 'b117' ORDER BY am.bid_num DESC ;

  CREATE OR REPLACE FUNCTION "J20201201"."FIRST_PDIMAGE" (VPD_NUM BOARD.PD_NUM%TYPE)
RETURN VARCHAR2
IS
  VPD_IMAGE BOARD.PD_IMAGE%TYPE;
  V_STATE NUMBER;
BEGIN
  SELECT COUNT(*)
  INTO V_STATE
  FROM BOARD
  WHERE PD_NUM = VPD_NUM AND (-1 != INSTR(PD_IMAGE, ','));
  
  IF V_STATE <= 0 THEN
    SELECT PD_IMAGE
    INTO VPD_IMAGE
    FROM BOARD
    WHERE PD_NUM = VPD_NUM;
    RETURN VPD_IMAGE;
  ELSE
    SELECT SUBSTR(PD_IMAGE,0,(INSTR(PD_IMAGE, ',')-1))
    INTO VPD_IMAGE
    FROM BOARD
    WHERE PD_NUM = VPD_NUM;
    RETURN VPD_IMAGE;
  END IF;
END;
